REDIM a$(1000000)
OPEN "D:\Dropbox\MSCollector Items\[0list.txt" FOR INPUT AS #1
DO
    a = a + 1
    LINE INPUT #1, a$(a)
LOOP UNTIL EOF(1) = -1
CLOSE #1
REDIM b$(1000000)
FOR b = 1 TO a
    OPEN "D:\Dropbox\MSCollector Items\" + a$(b) + ".txt" FOR INPUT AS #1
    LINE INPUT #1, b$
    IF b$ <> "" THEN
        c = c + 1
        b$(c) = b$
    END IF
    CLOSE #1
NEXT b

REDIM eqpvac$(32768)
REDIM usevac$(32768)
REDIM etcvac$(32768)
REDIM stpvac$(32768)
OPEN fileloc$ + "D:\Dropbox\MSCollector Items\[0eqp.txt" FOR INPUT AS #1

    eqpvac = eqpvac + 1
    LINE INPUT #1, eqpvac$(eqpvac)
LOOP UNTIL EOF(1) = -1
CLOSE #1
OPEN fileloc$ + "D:\Dropbox\MSCollector Items\[0use.txt" FOR INPUT AS #1

    usevac = usevac + 1
    LINE INPUT #1, usevac$(usevac)

    etcvac = etcvac + 1
    LINE INPUT #1, etcvac$(etcvac)
LOOP UNTIL EOF(1) = -1

DO
    stpvac = stpvac + 1
    LINE INPUT #1, stpvac$(stpvac)
LOOP UNTIL EOF(1) = -1
CLOSE #1

FOR e = 1 TO c
    FOR d = 1 TO eqpvac
        IF eqpvac$(d) = b$(e) THEN eqpvac$(d) = ""
    NEXT d
    FOR d = 1 TO eqpvac
        IF usevac$(d) = b$(e) THEN usevac$(d) = ""
    NEXT d
    FOR d = 1 TO etcvac
        IF etcvac$(d) = b$(e) THEN etcvac$(d) = ""
    NEXT d
    FOR d = 1 TO stpvac
        IF stpvac$(d) = b$(e) THEN stpvac$(d) = ""
    NEXT d
NEXT e
